017package org.microbean.narayana.jta.openwebbeans;
018
019import javax.enterprise.event.TransactionPhase;
020
021import javax.enterprise.inject.Instance;
022
023import javax.enterprise.inject.spi.CDI;
024import javax.enterprise.inject.spi.ObserverMethod;
025
026import javax.transaction.Transaction;
027import javax.transaction.TransactionManager;
028import javax.transaction.UserTransaction;
029import javax.transaction.SystemException;
030
031import org.apache.webbeans.ee.event.TransactionalEventNotifier;
032
033import org.apache.webbeans.spi.TransactionService;
034
035public class NarayanaTransactionService implements TransactionService {
036
037  public NarayanaTransactionService() {
038    super();
039  }
040
041  @Override
042  public final TransactionManager getTransactionManager() {
043    Instance<TransactionManager> transactionManagerInstance = CDI.current().select(TransactionManager.class);
044    assert transactionManagerInstance != null;
045    final TransactionManager transactionManager;
046    if (transactionManagerInstance.isUnsatisfied()) {
047      transactionManager = com.arjuna.ats.jta.TransactionManager.transactionManager();
048    } else {
049      transactionManager = transactionManagerInstance.get();
050    }
051    return transactionManager;
052  }
053
054  @Override
055  public final Transaction getTransaction() {
056    final Instance<Transaction> transactionInstance = CDI.current().select(Transaction.class);
057    Transaction transaction = null;
058    if (transactionInstance.isUnsatisfied()) {
059      final TransactionManager transactionManager = this.getTransactionManager();
060      if (transactionManager != null) {
061        try {
062          transaction = transactionManager.getTransaction();
063        } catch (final SystemException e) {
064          throw new RuntimeException(e.getMessage(), e);
065        }
066      }
067    } else {
068      transaction = transactionInstance.get();
069    }
070    return transaction;
071  }
072
073  /**
074   * Returns the {@link UserTransaction} present in this environment
075   * by invoking the {@link
076   * com.arjuna.ats.jta.UserTransaction#userTransaction()} method and
077   * returning its result.
078   *
079   * <p>This method never returns {@code null}.</p>
080   *
081   * <p>The return value of this method is used as the backing
082   * implementation of the <a
083   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#additional_builtin_beans">built-in
084   * {@code UserTransaction} CDI bean</a>.</p>
085   *
086   * @return the non-{@code null} {@link UserTransaction} present in
087   * this environment
088   *
089   * @see com.arjuna.ats.jta.UserTransaction#userTransaction()
090   */
091  @Override
092  public final UserTransaction getUserTransaction() {
093    return com.arjuna.ats.jta.UserTransaction.userTransaction();
094  }
095
096  @Override
097  public final void registerTransactionSynchronization(final TransactionPhase phase,
098                                                       final ObserverMethod<? super Object> observer,
099                                                       final Object event)
100    throws Exception {
101    TransactionalEventNotifier.registerTransactionSynchronization(phase, observer, event, null /* we aren't handed metadata, so none to pass */);
102  }
103  
104}
